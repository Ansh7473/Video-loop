<body>

    <video id="videoPlayer" controls autoplay></video>

    <script>

        const videos = [

            "https://raw.githubusercontent.com/Ansh7473/Video-loop/main/f5a42d5a.mp4",

            "https://raw.githubusercontent.com/Ansh7473/Video-loop/main/b99327e1.mp4",

            "https://raw.githubusercontent.com/Ansh7473/Video-loop/main/9d07a5bd.mp4"

        ];

       

        let currentVideo = 0;

        const videoPlayer = document.getElementById("videoPlayer");

        function playNextVideo() {

            videoPlayer.src = videos[currentVideo];

            videoPlayer.play();

            currentVideo = (currentVideo + 1) % videos.length; // Loop back to first video

        }

        videoPlayer.addEventListener("ended", playNextVideo);

        playNextVideo();

    </script>
